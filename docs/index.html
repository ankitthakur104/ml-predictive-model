# Model comparison — 5-fold cross-validation
  models = {
      "RandomForest":       RandomForestClassifier(n_estimators=100),
      "GradientBoosting":   GradientBoostingClassifier(n_estimators=100),
      "LogisticRegression": LogisticRegression(max_iter=1000),
  }
  best_model, best_score = None, 0
  for name, model in models.items():
      cv = cross_val_score(model, X_train, y_train, cv=5)
      print(f"{name}: {cv.mean():.4f} ± {cv.std():.4f}")
      if cv.mean() > best_score: best_model, best_score = (name, model), cv.mean()

  # Hyperparameter tuning on winner
  gs = GridSearchCV(best_model[1], param_grid, cv=3, n_jobs=-1)
  gs.fit(X_train, y_train)
  joblib.dump(gs.best_estimator_, "model.joblib")
